<script>
var map;
function init(){
// initiate leaflet map
map = new L.Map('cartodb-map', {
center: [19.8649719913073,-96.8427718266555],
zoom: 9
})

L.tileLayer('http://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}.png', {
attribution: '&copy; <a href="http://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors, &copy; <a href="http://cartodb.com/attributions">CartoDB</a>'
}).addTo(map);

var mapMinZoom = 13;
var mapMaxZoom = 17;
var mapBounds = new L.LatLngBounds(
new L.LatLng(45.51042998, 9.09105585),
new L.LatLng(45.52567771, 9.11608562));
L.tileLayer('http://cdb_tilestorage.storage.googleapis.com/dg_cust_sample_anp_combo_maptiler/{z}/{x}/{y}.png', {
minZoom: mapMinZoom, maxZoom: mapMaxZoom,
//bounds: mapBounds,
opacity: 1
}).addTo(map);

//area of interest from customer
var layerUrl = 'https://team.cartodb.com/u/athompson/api/v2_1/viz/c6e66768-0000-11e5-8be2-0e018d66dc29/viz.json';

cartodb.createLayer(map, layerUrl)
.addTo(map)
.on('done', function(layer) {
}).on('error', function() {
//log the error
});
}
</script>
</head>

<body onload="init()">
<div id='cartodb-map'></div>
